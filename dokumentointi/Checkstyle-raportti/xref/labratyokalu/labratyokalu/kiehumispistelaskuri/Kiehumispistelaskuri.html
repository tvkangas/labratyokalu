8   /**
9    * Luokka luo kiehumispistelaskurin
10   */
11  public class Kiehumispistelaskuri {
12  
13      private static final double kaasuvakio = 8.31446;
14      private double alkupaine;
15      private double alkulampotila;
16      private double loppupaine;
17      private double hoyrynpaine;
18  
19      public Kiehumispistelaskuri(double alkupaine, double alkulampotila, double loppupaine, double hoyrynpaine) {
20          this.alkupaine = alkupaine;
21          this.alkulampotila = alkulampotila;
22          this.loppupaine = loppupaine;
23          this.hoyrynpaine = hoyrynpaine;
24      }
25  
26      public double getKaasuvakio() {
27          return this.kaasuvakio;
28      }
29  
30      /**
31       * Metodi palauttaa arvon käänteisluvun
32       *
33       * @param arvo luku, josta käänteisluku halutaan
34       * @return käänteisluku
35       */
36      public double palautaKaanteisarvo(double arvo) {
37          return 1 / arvo;
38      }
39  
40      /**
41       * Metodi palauttaa luvun luonnollisen logaritmin
42       *
43       * @return luonnollinen logaritmi
44       */
45      public double palautaLuonnollinenLogaritmi(double luku) {
46          return Math.log(luku);
47      }
48  
49      /**
50       * Metodi palauttaa uuden kiehumispisteen
51       *
52       * @return uusi kiehusmipiste
53       */
54      public double laskeUusiKiehumispiste() {
55          double uusiKiehumispiste = palautaKaanteisarvo(-palautaLuonnollinenLogaritmi(this.loppupaine / this.alkupaine) * palautaKaasuvakionJaHoyrynpaineenSuhde()
56                  + palautaKaanteisarvo(muutaCelsiusKelviniksi(this.alkulampotila)));
57          return muutaKelvinCelsiukseksi(uusiKiehumispiste);
58      }
59  
60      /**
61       * Metodi palauttaa kaasuvakion ja höyrynpaineen suhteen R/dH
62       *
63       * @return suhde R/dH
64       */
65      public double palautaKaasuvakionJaHoyrynpaineenSuhde() {
66          return getKaasuvakio() / this.hoyrynpaine;
67      }
68  
69      /**
70       * Metodi muuttaa Celsius-asteet Kelvineiksi
71       *
72       * @param luku muutettava luku
73       * @return muutettu luku
74       */
75      public double muutaCelsiusKelviniksi(double luku) {
76          return luku + 273.15;
77      }
78  
79      /**
80       * Metodi muuttaa Kelvin-asteeet Celsius-asteeksi
81       *
82       * @param luku muutettava luku
83       * @return muutettu luku
84       */
85      public double muutaKelvinCelsiukseksi(double luku) {
86          return luku - 273.15;
87      }
88  
89  }
