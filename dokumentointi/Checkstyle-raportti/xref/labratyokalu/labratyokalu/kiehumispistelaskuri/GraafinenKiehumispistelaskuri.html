1   package labratyokalu.labratyokalu.kiehumispistelaskuri;
2   
3   import java.awt.*;
4   import java.awt.event.*;
5   import javax.swing.*;
6   
7   /**
8    * @author Tuukka Kangas
9    * @version 1.0
10   * @since 2015-09-01
11   */
12  /**
13   * Luokka luo kiehumispistelaskurin, jota voi käyttää graafisesti
14   */
15  public class GraafinenKiehumispistelaskuri implements Runnable {
16  
17      private JFrame frame;
18      private JTextField alkupaine;
19      private JTextField loppupaine;
20      private JTextField alkukiehumispiste;
21      private JTextField hoyrynpaine;
22      private JTextField loppukiehumispiste;
23  
24      public GraafinenKiehumispistelaskuri() {
25          this.alkupaine = new JTextField("");
26          this.loppupaine = new JTextField("");
27          this.alkukiehumispiste = new JTextField("");
28          this.hoyrynpaine = new JTextField("406600.0");
29          this.loppukiehumispiste = new JTextField("");
30      }
31  
32      @Override
33      public void run() {
34          frame = new JFrame("Kiehumispistelaskuri");
35          frame.setPreferredSize(new Dimension(600, 300));
36          frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
37  
38          luoKomponentit(frame.getContentPane());
39          frame.pack();
40          frame.setVisible(true); 
41      }
42      
43      private void luoKomponentit(Container container) {
44          GridLayout layout = new GridLayout(2, 2);
45          container.setLayout(layout);
46          alustaAputekstit(container);
47          alustaNaytot(container);
48          JButton muuntoNappain = new JButton("Laske uusi kiehumispiste");
49          muuntoNappain.addActionListener(new KlikkauskuuntelijaMuuntonappain(this.alkupaine, this.loppupaine, this.alkukiehumispiste, this.hoyrynpaine, this.loppukiehumispiste));
50         
51          container.add(muuntoNappain);
52          container.add(new JLabel("<html>Paineet on annettava samassa yksikössä. <br /><br /> Höyrynpaineita [J/mol] <br /> Vesi 406600.0 <br /> Butaani 210000.0 <br /> Asetoni 313000.0</html>"));
53          asetaNayttojenKaytettavyys();
54      }
55      
56      private void alustaAputekstit(Container c) {
57          JPanel apuPanel = new JPanel(new GridLayout(5, 1));
58          JLabel alkupaineApu = new JLabel("Syötä alkuperäinen paine");
59          JLabel loppupaineApu = new JLabel("Syötä haluttu paine");
60          JLabel alkulampotilaApu = new JLabel("Syötä kiehumispiste tiedetyssä paineessa [C]");
61          JLabel hoyrynpaineApu = new JLabel("Syötä höyrynpaine [J/mol]");
62          JLabel loppulampotilaApu = new JLabel("Kiehumispiste uudessa paineessa [C]");
63          apuPanel.add(alkupaineApu);
64          apuPanel.add(loppupaineApu);
65          apuPanel.add(alkulampotilaApu);
66          apuPanel.add(hoyrynpaineApu);
67          apuPanel.add(loppulampotilaApu);
68          c.add(apuPanel);
69      }
70      
71      private void alustaNaytot(Container c) {
72          JPanel apuPanel = new JPanel(new GridLayout(5, 1));
73          apuPanel.add(this.alkupaine);
74          apuPanel.add(this.loppupaine);
75          apuPanel.add(this.alkukiehumispiste);
76          apuPanel.add(this.hoyrynpaine);
77          apuPanel.add(this.loppukiehumispiste);
78          
79          c.add(apuPanel);
80      }
81      
82      private void asetaNayttojenKaytettavyys() {
83          this.loppukiehumispiste.setEnabled(false);
84      }
85  
86  }
