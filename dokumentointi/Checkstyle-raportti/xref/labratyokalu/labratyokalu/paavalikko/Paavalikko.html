22  /**
23   * Luokka luo päävalikon, jossa voi selata haluttuja toimintoja
24   */
25  
26  public class Paavalikko implements Runnable {
27      private JFrame frame;
28      private MuistiinpanotPaavalikko muistiinpanotPaavalikko;
29  
30      public Paavalikko() {
31          try {
32              this.muistiinpanotPaavalikko = new MuistiinpanotPaavalikko();
33          } catch (Exception ex) {
34              Logger.getLogger(Paavalikko.class.getName()).log(Level.SEVERE, null, ex);
35          }   
36      }
37      
38      @Override
39      public void run() {
40          frame = new JFrame("Päävalikko");
41          frame.setPreferredSize(new Dimension(300, 400));
42          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
43          luoKomponentit(frame.getContentPane());
44          frame.pack();
45          frame.setVisible(true);        
46      }
47          
48      private void luoKomponentit(Container container) {
49          GridLayout layout = new GridLayout(1, 1);
50          container.setLayout(layout);
51          alustaValikkonappaimet(container);
52      }    
53      
54      private void alustaValikkonappaimet(Container container) {
55          JPanel apupaneeli = new JPanel(new GridLayout(6, 1));
56          JButton mpNappain = new JButton("Muistiinpanot");
57          mpNappain.addActionListener(new KlikkauskuuntelijaMuistiinpanot(muistiinpanotPaavalikko));
58          JButton laskinNappain = new JButton("Laskin");
59          laskinNappain.addActionListener(new KlikkauskuuntelijaLaskin(new GraafinenLaskin()));
60          JButton yksikkomuunninNappain = new JButton("Yksikkömuunnin");
61          yksikkomuunninNappain.addActionListener(new KlikkauskuuntelijaYksikkomuunnin(new GraafinenYksikkomuunninValikko()));
62          JButton moolimassalaskuriNappain = new JButton("Moolimassalaskuri");
63          moolimassalaskuriNappain.addActionListener(new KlikkauskuuntelijaMoolimassalaskuri(new GraafinenMoolimassalaskuri()));
64          JButton ajastinNappain = new JButton("Ajastin");
65          ajastinNappain.addActionListener(new KlikkauskuuntelijaAjastin(new AjastimenValikko()));
66          JButton kiehumispisteNappain = new JButton("Kiehumispistelaskuri");
67          kiehumispisteNappain.addActionListener(new KlikkauskuuntelijaKiehumispistelaskuri(new GraafinenKiehumispistelaskuri()));
68          
69          apupaneeli.add(mpNappain);
70          apupaneeli.add(laskinNappain);
71          apupaneeli.add(yksikkomuunninNappain);
72          apupaneeli.add(moolimassalaskuriNappain);
73          apupaneeli.add(ajastinNappain);
74          apupaneeli.add(kiehumispisteNappain);
75          container.add(apupaneeli);
76      }
77      
78  }
