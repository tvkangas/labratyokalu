18  /**
19   * Luokan avulla luodaan käyttöliittymä muistiinpanojen lisäystä varten
20   */
21  
22  public class GraafinenKayttoliittymaMuistiinpanonLisays implements Runnable {
23  
24      private JFrame frame;
25      private MuistiinpanoVarasto muistiinpanoVarasto;
26  
27      public GraafinenKayttoliittymaMuistiinpanonLisays(MuistiinpanoVarasto muistiinpanoVarasto) {
28          this.muistiinpanoVarasto = muistiinpanoVarasto;
29      } 
30  
31      @Override
32      public void run() {
33          frame = new JFrame("Muistiinpanon lisäys");
34          frame.setPreferredSize(new Dimension(550, 200));
35          frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
36          luoKomponentit(frame.getContentPane());
37          frame.pack();
38          frame.setVisible(true);
39      }
40      
41      /**
42       * Metodi luo halutut komponentit käyttöliittymään
43       *
44       * @param container Piirrossa hyödynnetty säiliö, johon tallenetaan halutut asiat
45       * 
46       */
47  
48      private void luoKomponentit(Container container) {
49          GridLayout layout = new GridLayout(3, 2);
50          container.setLayout(layout);
51  
52          JLabel paivamaaraTeksti = new JLabel("Päivämäärä: ");
53          JTextField paivamaaraKentta = new JTextField(haeTamaPaiva());
54  
55          JLabel muistiinpanoTeksti = new JLabel("Muistiinpano: ");
56          JTextField muistiinpanoKentta = new JTextField();
57  
58          JButton tallennusNappi = new JButton("Lisää muistiinpano");
59          tallennusNappi.addActionListener(new KlikkauskuuntelijaMuistiinpanonLisays(this.muistiinpanoVarasto, paivamaaraKentta, muistiinpanoKentta, this.frame));
60  
61          container.add(paivamaaraTeksti);
62          container.add(paivamaaraKentta);
63          container.add(muistiinpanoTeksti);
64          container.add(muistiinpanoKentta);
65          container.add(new JLabel(""));
66          container.add(tallennusNappi);
67      }
68      
69      /**
70       * Metodi tuo halutun päivämäärän hyödyntäen Javan Date-metodia
71       *
72       * @return päivämäärän merkkijonona
73       * 
74       */
75  
76      private String haeTamaPaiva() {
77          Date dNow = new Date();
78          SimpleDateFormat ft = new SimpleDateFormat("dd.MM.yyyy");
79          return ft.format(dNow);
80      }
81  
82  }
