15  /**
16   * Luokka luo pohjan, jossa voidaan selata eri vaihtoehtoja muistiinpanojen
17   * tekemisessä
18   */
19  public class MuistiinpanotPaavalikko implements Runnable {
20  
21      private JFrame frame;
22      private MuistiinpanoVarasto muistiinpanoVarasto;
23      private GraafinenKayttoliittymaMuistiinpanonLisays lisaysValikko;
24      private GraafinenKayttoliittymaMuistiinpanojenTarkastelu tarkasteluValikko;
25  
26      public MuistiinpanotPaavalikko() throws Exception {
27          this.muistiinpanoVarasto = luoMuistiinpanoVarasto();
28          this.lisaysValikko = new GraafinenKayttoliittymaMuistiinpanonLisays(this.muistiinpanoVarasto);
29          this.tarkasteluValikko = new GraafinenKayttoliittymaMuistiinpanojenTarkastelu(this.muistiinpanoVarasto);
30      }
31  
32      @Override
33      public void run() {
34          frame = new JFrame("Muistiinpano");
35          frame.setPreferredSize(new Dimension(550, 200));
36          frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
37          luoKomponentit(frame.getContentPane());
38          frame.pack();
39          frame.setVisible(true);
40      }
41  
42      private void luoKomponentit(Container container) {
43          GridLayout layout = new GridLayout(1, 1);
44          container.setLayout(layout);
45          alustaValintanappaimet(container);
46      }
47  
48      private void alustaValintanappaimet(Container c) {
49          JPanel panelApu = new JPanel(new GridLayout(2, 1));
50          JButton lisaysNappain = new JButton("Lisää muistiinpano");
51          lisaysNappain.addActionListener(new KlikkauskuuntelijaPaavalikkoLisaaMuistiinpano(lisaysValikko, lisaysNappain));
52  
53          JButton tarkasteluNappain = new JButton("Tarkastele muistiinpanoja");
54          tarkasteluNappain.addActionListener(new KlikkauskuuntelijaPaavalikkoTarkasteleMuistiinpanoja(tarkasteluValikko, tarkasteluNappain));
55  
56          panelApu.add(lisaysNappain);
57          panelApu.add(tarkasteluNappain);
58          c.add(panelApu);
59      }
60  
61      /**
62       * Metodi, joka lukee tallennetut muistiinpanot
63       *
64       * @return apuVarasto joka palauttaa konstruktorille luetuista
65       * muistiinpanoista koostuvan listan
66       *
67       */
68      public MuistiinpanoVarasto luoMuistiinpanoVarasto() throws Exception {
69          MuistiinpanoVarasto apuVarasto = new MuistiinpanoVarasto();
70          File tiedosto = new File("src/main/java/labratyokalu/tiedostot/muistiinpanot.txt");
71          Scanner skanneri = new Scanner(tiedosto, "UTF-8");
72          while (skanneri.hasNextLine()) {
73              Scanner apu = new Scanner(skanneri.nextLine());
74              apu.useDelimiter(";");
75              String mp = apu.next();
76              String jattopaiva = apu.next();
77              apuVarasto.lisaaMuistiinpano(new Muistiinpano(mp, jattopaiva));
78          }
79          return apuVarasto;
80      }
81  }
