12  /**
13   * Luokka luo graafisen yksikkömuuntimen, joka koskee massan yksikkömuunnoksia
14   */
15  public class GraafinenYksikkomuunninMassa implements Runnable {
16  
17      private JFrame frame;
18      private JTextField vasen;
19      private JTextField oikea;
20  
21      public GraafinenYksikkomuunninMassa() {
22          this.vasen = new JTextField("");
23          this.oikea = new JTextField("");
24      }
25  
26      @Override
27      public void run() {
28          frame = new JFrame("Massamuunnin");
29          frame.setPreferredSize(new Dimension(500, 550));
30          frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
31  
32          luoKomponentit(frame.getContentPane());
33          frame.pack();
34          frame.setVisible(true);
35      }
36  
37      private void luoKomponentit(Container container) {
38          GridLayout layout = new GridLayout(3, 1);
39          container.setLayout(layout);
40          alustaAputekstit(container);
41          alustaNaytot(container);
42  
43          alustaMuuntoNappaimet(container);
44  
45      }
46  
47      private void alustaAputekstit(Container c) {
48          JPanel panelApu = new JPanel(new GridLayout(1, 2));
49          JLabel vasen = new JLabel("Syötä tähän lukusi");
50          JLabel oikea = new JLabel("Tässä näet muutetun luvun");
51          panelApu.add(vasen);
52          panelApu.add(oikea);
53          c.add(panelApu);
54      }
55  
56      private void alustaNaytot(Container container) {
57          JPanel panel1 = new JPanel(new GridLayout(1, 2));
58          panel1.add(this.vasen);
59          panel1.add(this.oikea);
60          container.add(panel1);
61          asetaNayttojenKaytettavyys();
62      }
63  
64      private void alustaMuuntoNappaimet(Container c) {
65          JPanel panel2 = new JPanel(new GridLayout(1, 2));
66  
67          JButton kgTolbNappi = new JButton("kg >> lb");
68          kgTolbNappi.addActionListener(new KlikkausKuuntelijaMuunnin(new Yksikkomuunnin(2.2046, 0, "kg", "lb"), this.vasen, this.oikea));
69          panel2.add(kgTolbNappi);
70  
71          JButton lbToKgNappi = new JButton("lb >> kg");
72          lbToKgNappi.addActionListener(new KlikkausKuuntelijaMuunnin(new Yksikkomuunnin((1.0 / 2.2046), 0, "lb", "kg"), this.vasen, this.oikea));
73          panel2.add(lbToKgNappi);
74  
75          c.add(panel2);
76      }
77  
78      private void asetaNayttojenKaytettavyys() {
79          this.vasen.setEnabled(true);
80          this.oikea.setEnabled(false);
81      }
82  
83  }
