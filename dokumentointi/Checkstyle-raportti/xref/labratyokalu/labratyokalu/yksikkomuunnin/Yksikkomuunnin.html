3   public class Yksikkomuunnin {
4   
5       double kerroin;
6       double vakiotermi;
7       String mista;
8       String mihin;
9   
10      /**
11       * @author Tuukka Kangas
12       * @version 1.0
13       * @since 2015-08-18
14       */
15      /**
16       * Luokka luo Yksikkömuunnin-olioita, joiden avulla voidaan suorittaa
17       * yksikkömuunnoksia sekä kertoimilla että vakiotermein
18       */
19      public Yksikkomuunnin(double k, double vk, String mista, String mihin) {
20          this.kerroin = k;
21          this.vakiotermi = vk;
22          this.mista = mista;
23          this.mihin = mihin;
24      }
25  
26      /**
27       * Metodi muuntaa luvun annettujen oliomuuttujien mukaan
28       * @param muunnettava metodi saa parametrin arvona muunnettava luvun
29       * @return palauttaa muunnetun luvun
30       */
31      public double muunna(double muunnettava) {
32          return muunnettava * this.kerroin + this.vakiotermi;
33      }
34  
35      public String getYksikko() {
36          return this.mihin;
37      }
38  
39      public String getMista() {
40          return this.mista;
41      }
42  
43      public double getKerroin() {
44          return this.kerroin;
45      }
46  
47  }
