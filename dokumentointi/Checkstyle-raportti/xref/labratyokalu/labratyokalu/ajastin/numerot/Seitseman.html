12  /**
13   * Luokka luo Numero luokan toteuttavan olion, joka piirtää numeron seitsemän
14   */
15  
16  public class Seitseman extends Numero {
17  
18      public Seitseman(int x, int y, int koko, Color vari) {
19          super(x, y, koko, vari);
20      }
21      
22      @Override
23      public void piirra(Graphics graphics) {
24          graphics.setColor(vari);
25          graphics.fillRect(getAlkuX(), getAlkuY(), getKoko(), getKoko());
26          graphics.fillRect(getKoko() + getAlkuX() + 1, getAlkuY(), getKoko(), getKoko());
27          graphics.fillRect(2 * getKoko() + getAlkuX() + 2, getAlkuY(), getKoko(), getKoko());
28          graphics.fillRect(2 * getKoko() + getAlkuX() + 2, getKoko() + getAlkuY() + 1, getKoko(), getKoko());
29          graphics.fillRect(2 * getKoko() + getAlkuX() + 2, 2 * getKoko() + getAlkuY() + 2, getKoko(), getKoko());
30          graphics.fillRect(2 * getKoko() + getAlkuX() + 2, 3 * getKoko() + getAlkuY() + 3, getKoko(), getKoko());
31          graphics.fillRect(2 * getKoko() + getAlkuX() + 2, 4 * getKoko() + getAlkuY() + 4, getKoko(), getKoko());
32  
33      }
34  
35  }
