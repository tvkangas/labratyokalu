27      public GraafinenAjastin(Ajastin ajastin) {
28          this.ajastin = ajastin;
29          this.numerosarja = new Numerosarja(this.ajastin);
30          this.piirtoalusta = new Piirtoalusta(this.numerosarja);
31          this.aLogiikka = new AjastinLogiikka(this.ajastin, this.piirtoalusta, this.numerosarja);
32      }

36          frame = new JFrame();

38  
39          frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
40  
41          luoKomponentit(frame.getContentPane());
42          lueAjastin();
43  
44          frame.pack();
45          frame.setVisible(true);
46      }

48      private void luoKomponentit(Container container) {
49          GridLayout layout = new GridLayout(1, 1);
50          JPanel paaPanel = new JPanel(new GridLayout(2, 1));
51          paaPanel.add(piirtoalusta);
52          paaPanel.add(lisaaNapit(container));
53          container.add(paaPanel);
54      }

56      private void lueAjastin() {
57          int koko = 20;
58          Color vari = Color.RED;
59          this.numerosarja.lisaaNumero(muutaNumeroksi(this.ajastin.getTunnitKymmenet(), 20, 20, koko, vari));
60          this.numerosarja.lisaaNumero(muutaNumeroksi(this.ajastin.getTunnitYkkoset(), 90, 20, koko, vari));
61          this.numerosarja.lisaaNumero(new Kaksoispiste(140, 20, koko, vari));
62          this.numerosarja.lisaaNumero(muutaNumeroksi(this.ajastin.getMinuutitKymmenet(), 190, 20, koko, vari));
63          this.numerosarja.lisaaNumero(muutaNumeroksi(this.ajastin.getMinuutitYkkoset(), 260, 20, koko, vari));
64          this.numerosarja.lisaaNumero(new Kaksoispiste(310, 20, koko, vari));
65          this.numerosarja.lisaaNumero(muutaNumeroksi(this.ajastin.getSekunnitKymmenet(), 360, 20, koko, vari));
66          this.numerosarja.lisaaNumero(muutaNumeroksi(this.ajastin.getSekunnitYkkoset(), 430, 20, koko, vari));
67      }

69      /**
70       * Metodi muuttaa luvun piirrettäväksi numeroksi
71       * 
72       * @param luku Integer muodossa
73       * @param x x-koordinaatti, josta piirtäminen aloitetaan
74       * @param y y-koordinaatti, josta piirtäminen aloitetaan
75       * @param koko yhden laatikon koko
76       * @param vari laatikon väri
77       * @return palauttaa Numeron
78       */
79      
80      public Numero muutaNumeroksi(int luku, int x, int y, int koko, Color vari) {
81          if (luku == 1) {
82              return new Yksi(x, y, koko, vari);
83          } else if (luku == 2) {
84              return new Kaksi(x, y, koko, vari);
85          } else if (luku == 3) {
86              return new Kolme(x, y, koko, vari);
87          } else if (luku == 4) {
88              return new Nelja(x, y, koko, vari);
89          } else if (luku == 5) {
90              return new Viisi(x, y, koko, vari);
91          } else if (luku == 6) {
92              return new Kuusi(x, y, koko, vari);
93          } else if (luku == 7) {
94              return new Seitseman(x, y, koko, vari);
95          } else if (luku == 8) {
96              return new Kahdeksan(x, y, koko, vari);
97          } else if (luku == 9) {
98              return new Yhdeksan(x, y, koko, vari);
99          } else if (luku == 0) {
100             return new Nolla(x, y, koko, vari);
101         } else {
102             return null;
103         }
104     }

106     private JPanel lisaaNapit(Container c) {
107         JPanel panelApu = new JPanel(new GridLayout(1, 1));
108         panelApu.setSize(new Dimension(100, 150));
109         JButton kaynnista = new JButton("Käynnistä");
110         kaynnista.setBackground(Color.BLACK);
111         kaynnista.setForeground(Color.RED);
112         kaynnista.setOpaque(true);
113         kaynnista.addActionListener(new KlikkauskuuntelijaKaynnista(this.aLogiikka, this.piirtoalusta));
114 
115         panelApu.add(kaynnista);
116         return panelApu;
117     }
