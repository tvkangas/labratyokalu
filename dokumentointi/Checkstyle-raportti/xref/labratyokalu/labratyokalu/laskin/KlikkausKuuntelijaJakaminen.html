15  /**
16   * Luokka luo ActionListener-olion, jolla voidaan jakaa
17   */
18  
19  public class KlikkausKuuntelijaJakaminen implements ActionListener {
20  
21      private Laskin laskin;
22      private JTextField tekstikentta;
23      private JTextField tuloskentta;
24      private JButton nappi;
25  
26      public KlikkausKuuntelijaJakaminen(Laskin laskin, JTextField tekstikentta, JTextField tuloskentta, JButton nappi) {
27          this.laskin = laskin;
28          this.tekstikentta = tekstikentta;
29          this.tuloskentta = tuloskentta;
30          this.nappi = nappi;
31      }
32  
33      @Override
34      public void actionPerformed(ActionEvent e) {
35          String teksti = tekstikentta.getText();
36          try {
37              double luku = Double.parseDouble(teksti);
38              this.laskin.jaa(luku);
39              this.tuloskentta.setText("" + this.laskin.getArvo());
40              this.tekstikentta.setText("");
41          } catch (NumberFormatException ei) {
42              this.tekstikentta.setText("");
43          }
44  
45          String teksti2 = this.tuloskentta.getText();
46          if (!teksti2.equals("0")) {
47              this.nappi.setEnabled(true);
48          } else {
49              this.nappi.setEnabled(false);
50          }
51          
52      }
53  
54  }
