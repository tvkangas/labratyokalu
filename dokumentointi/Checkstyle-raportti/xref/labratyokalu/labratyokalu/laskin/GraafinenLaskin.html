14  /**
15   * Luokka luo graafisen laskimen, joka hyödyntää laskin-oliota
16   */
17  
18  public class GraafinenLaskin implements Runnable {
19  
20      private JFrame frame;
21      private Laskin laskin;
22      private JTextField yla;
23      private JTextField keski;
24  
25      public GraafinenLaskin() {
26          this.laskin = new Laskin();
27          this.yla = new JTextField(this.laskin.getArvo() + "");
28          this.keski = new JTextField("");
29      }
30  
31      @Override
32      public void run() {
33          frame = new JFrame("Laskin");
34          frame.setPreferredSize(new Dimension(300, 250));
35          frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
36  
37          luoKomponentit(frame.getContentPane());
38          frame.pack();
39          frame.setVisible(true);
40          this.keski.requestFocus();
41      }
42  
43      private void luoKomponentit(Container container) {
44          GridLayout layout = new GridLayout(4, 1);
45          container.setLayout(layout);
46          yla.setEnabled(false);        
47          container.add(yla);
48          
49          container.add(keski);
50          alustaNapit(container, yla, keski);
51      }
52  
53      public JFrame getFrame() {
54          return frame;
55      }    
56  
57      private void alustaNapit(Container container, JTextField yla, JTextField keski) {
58  
59          JPanel panel1 = new JPanel(new GridLayout(1, 3));
60          JButton plus = new JButton("+");
61          JButton miinus = new JButton("-");
62          JButton tyhjennys = new JButton("Z");
63  
64          plus.addActionListener(new KlikkausKuuntelijaSummaus(this.laskin, keski, yla, tyhjennys));
65          miinus.addActionListener(new KlikkausKuuntelijaVahennys(this.laskin, keski, yla, tyhjennys));
66          tyhjennys.addActionListener(new KlikkausKuuntelijaTyhjenna(this.laskin, keski, yla, tyhjennys));
67  
68          panel1.add(plus);
69          panel1.add(miinus);
70          panel1.add(tyhjennys);
71          container.add(panel1);
72  
73          JPanel panel2 = new JPanel(new GridLayout(1, 3));
74          JButton kertominen = new JButton("x");
75          JButton jakaminen = new JButton("/");
76          JButton korotaPotenssiin = new JButton("^");
77          
78          kertominen.addActionListener(new KlikkausKuuntelijaKertominen(this.laskin, keski, yla, tyhjennys));
79          jakaminen.addActionListener(new KlikkausKuuntelijaJakaminen(this.laskin, keski, yla, tyhjennys));
80          korotaPotenssiin.addActionListener(new KlikkausKuuntelijaPotenssi(this.laskin, keski, yla, tyhjennys));
81          
82          panel2.add(kertominen);
83          panel2.add(jakaminen);
84          panel2.add(korotaPotenssiin);
85          container.add(panel2);
86      }
87  
88  }
