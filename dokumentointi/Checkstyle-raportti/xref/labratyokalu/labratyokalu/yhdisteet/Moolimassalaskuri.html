10  /**
11   * Luokka luo moolimassalaskuri-olion. Sen avulla voidaan laskea jonkin
12   * yhdisteen moolimassa
13   */
14  public class Moolimassalaskuri {
15  
16      private HashMap<Alkuaine, Integer> yhdisteenAlkuaineet;
17  
18      public Moolimassalaskuri(HashMap<Alkuaine, Integer> alkuaineet) {
19          this.yhdisteenAlkuaineet = alkuaineet;
20      }
21  
22      /**
23       * Metodi suoritta kertolaskun
24       *
25       *
26       * @param kerroin, kuinka monta kertaa ko massa pitää kertoa
27       * @param kerrottavaMassa, kerrottava massa
28       * @return tulo
29       */
30      public double suoritaKertolasku(int kerroin, double kerrottavaMassa) {
31          return 1.0 * kerroin * kerrottavaMassa;
32      }
33  
34      /**
35       * Metodi suoritta kertolaskun
36       *
37       *
38       * @return moolimassa
39       */
40      public double palautaYhdisteenMassa() {
41          double moolimassa = 0;
42          for (Alkuaine aa : this.yhdisteenAlkuaineet.keySet()) {
43              moolimassa += suoritaKertolasku(this.yhdisteenAlkuaineet.get(aa), aa.getMassa());
44          }
45          return moolimassa;
46      }
47  
48  }
