12   */
13  public class Alkuaine {
14      private int jarjestysluku;
15      private String nimi;
16      private String lyhenne;
17      private double massa;
18      private double tiheys;    
19  
20      public Alkuaine(int jarjluku, String nimi, String lyhenne, double massa, double tiheys) {
21          this.jarjestysluku = jarjluku;
22          this.nimi = nimi;
23          this.lyhenne = lyhenne;
24          this.massa = massa;
25          this.tiheys = tiheys;
26      }
27      
28      public String getNimi() {
29          return this.nimi;
30      }
31  
32      public String getLyhenne() {
33          return lyhenne;
34      }
35  
36      public double getMassa() {
37          return massa;
38      }
39  
40      public double getTiheys() {
41          return tiheys;
42      }
43      
44      public int getJarjestysluku() {
45          return this.jarjestysluku;
46      }
47      
48      @Override
49      public String toString() {
50          return getJarjestysluku() + ". " + getNimi() + " (" + getLyhenne() + ") M = " + getMassa() + ", tiheys = " + getTiheys() + " x 10^3 kg/m^3";
51      }

53      
54  }
